# Prolly Kotlin Cookbook

The Kotlin binding is generated by UniFFI and uses package
`build.crab.prolly`. Call `ProllyNative.useLocalDebugLibrary()` in local tests
to load `target/debug/libprolly_bindings.*`.

Runnable scenarios live as separate files/classes under
`build.crab.prolly.examples`, matching the Rust example style:

```sh
cargo build -p prolly-bindings
mvn -q -f crates/prolly/bindings/kotlin/pom.xml \
  compile \
  -Dexec.mainClass=build.crab.prolly.examples.CookbookScenariosKt \
  exec:java
mvn -q -f crates/prolly/bindings/kotlin/pom.xml \
  compile \
  -Dexec.mainClass=build.crab.prolly.examples.BasicMapKt \
  exec:java
mvn -q -f crates/prolly/bindings/kotlin/pom.xml \
  compile \
  -Dexec.mainClass=build.crab.prolly.examples.SecondaryIndexKt \
  exec:java
```

Application-style files include `BatchBuild.kt`, `LocalFirstState.kt`,
`Resolver.kt`, `CrdtMerge.kt`, `ConversationMemory.kt`,
`AgentEventLog.kt`, `BackgroundCompaction.kt`,
`DeterministicRagSnapshot.kt`, `DocumentChunkIndex.kt`,
`VectorSidecar.kt`, `ProvenanceValues.kt`, `MaterializedView.kt`,
`FilesystemSnapshot.kt`, and `DurableSqlite.kt`.

## Create A Durable Index

```kotlin
import build.crab.prolly.*

fun main() {
    ProllyNative.useLocalDebugLibrary()

    ProllyEngine.sqlite("app.prolly.db", defaultConfig()).use { engine ->
        var tree = engine.create()
        tree = engine.batch(
            tree,
            listOf(
                MutationRecord(MutationKind.UPSERT, "user/1".toByteArray(), "Ada".toByteArray()),
                MutationRecord(MutationKind.UPSERT, "user/2".toByteArray(), "Linus".toByteArray()),
            ),
        )

        engine.publishNamedRoot("users/main".toByteArray(), tree)
    }
}
```

## Merge Writers

```kotlin
val base = tree
val left = engine.put(base, "user/1".toByteArray(), "Ada Lovelace".toByteArray())
val right = engine.put(base, "user/1".toByteArray(), "Countess Ada".toByteArray())

val merged = engine.merge(base, left, right, "prefer_right")

val callbackMerged = engine.mergeWithResolver(base, left, right, object : MergeResolverCallback {
    override fun resolve(conflict: ConflictRecord): ResolutionRecord =
        if (conflict.left != null && conflict.right != null) {
            ResolutionRecord(
                ResolutionKind.VALUE,
                conflict.left!! + " | ".toByteArray() + conflict.right!!,
            )
        } else {
            ResolutionRecord(ResolutionKind.UNRESOLVED, null)
        }
})
```

## Large Values And Blob GC

```kotlin
ProllyBlobStore.file("app.blobs").use { blobStore ->
    val large = ByteArray(1_000_000) { 120 }
    tree = engine.putLargeValue(
        blobStore,
        tree,
        "doc/1".toByteArray(),
        large,
        LargeValueConfigRecord(4096UL),
    )

    val loaded = engine.getLargeValue(blobStore, tree, "doc/1".toByteArray())
    val plan = engine.planBlobStoreGc(blobStore, listOf(tree))
    if (plan.reclaimableBlobCount > 0UL) {
        engine.sweepBlobStoreGc(blobStore, listOf(tree))
    }
}
```

## Custom Stores

Implement `HostStoreCallback` when Kotlin owns persistence. The callback must
return record values instead of throwing for expected store errors.

```kotlin
class MemoryHostStore : HostStoreCallback {
    private val nodes = linkedMapOf<List<Byte>, ByteArray>()

    override fun get(key: ByteArray): HostStoreBytesResultRecord =
        HostStoreBytesResultRecord(nodes[key.toList()]?.copyOf(), null)

    override fun put(key: ByteArray, value: ByteArray): HostStoreUnitResultRecord {
        nodes[key.toList()] = value.copyOf()
        return HostStoreUnitResultRecord(null)
    }

    override fun delete(key: ByteArray): HostStoreUnitResultRecord {
        nodes.remove(key.toList())
        return HostStoreUnitResultRecord(null)
    }

    override fun batch(ops: List<MutationRecord>): HostStoreUnitResultRecord {
        for (op in ops) {
            if (op.kind == MutationKind.UPSERT) nodes[op.key.toList()] = requireNotNull(op.value).copyOf()
            else nodes.remove(op.key.toList())
        }
        return HostStoreUnitResultRecord(null)
    }

    override fun batchGetOrdered(keys: List<ByteArray>) =
        HostStoreBatchGetResultRecord(keys.map { nodes[it.toList()]?.copyOf() }, null)

    override fun prefersBatchReads() = HostStoreBoolResultRecord(true, null)
    override fun supportsHints() = HostStoreBoolResultRecord(false, null)
    override fun getHint(namespace: ByteArray, key: ByteArray) = HostStoreBytesResultRecord(null, null)
    override fun putHint(namespace: ByteArray, key: ByteArray, value: ByteArray) = HostStoreUnitResultRecord(null)
    override fun listNodeCids() = HostStoreListBytesResultRecord(nodes.keys.map { ByteArray(it.size) { i -> it[i] } }, null)
    override fun getRoot(name: ByteArray) = HostStoreRootResultRecord(null, null)
    override fun putRoot(name: ByteArray, manifest: RootManifestRecord) = HostStoreUnitResultRecord(null)
    override fun deleteRoot(name: ByteArray) = HostStoreUnitResultRecord(null)
    override fun compareAndSwapRoot(name: ByteArray, expected: RootManifestRecord?, replacement: RootManifestRecord?) =
        HostStoreRootCasResultRecord(false, null, null)
    override fun listRoots() = HostStoreListRootsResultRecord(emptyList(), null)
}

val custom = ProllyEngine.customStore(MemoryHostStore(), defaultConfig())
```
